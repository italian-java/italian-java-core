<dependency>
    <groupId>com.github.italian-java</groupId>
    <artifactId>italian-java-core</artifactId>
    <version>1.0.0</version>
</dependency>


<repositories>
    <repository>
        <id>italian-java</id>

    </repository>
</repositories>
